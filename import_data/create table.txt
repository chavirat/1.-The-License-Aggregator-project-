CREATE TABLE tldr_license(
tldr_key VARCHAR(100) PRIMARY KEY,
tldr_name VARCHAR(200) NOT NULL,
tldr_url VARCHAR(300) NOT NULL,
summary LONGTEXT
);
CREATE TABLE tldr_requirements(
tldr_key VARCHAR(100) NOT NULL,
requirements VARCHAR(100) NOT NULL,
type VARCHAR(100) NOT NULL,
CONSTRAINT tldr_requirements_pk PRIMARY KEY (tldr_key,requirements,type)
);
CREATE TABLE spdx_license (
SPDX_key VARCHAR(100) PRIMARY KEY,
SPDX_fullname VARCHAR(200) NOT NULL,
SPDX_source_url VARCHAR(1000),
SPDX_notes LONGTEXT,
license_header LONGTEXT
);
CREATE TABLE nexb_license (
nexb_key VARCHAR(100) PRIMARY KEY,
nexb_name VARCHAR(200),
short_name VARCHAR(200),
guidance VARCHAR(1000),
category VARCHAR(200),
license_type VARCHAR(200),
license_profile VARCHAR(1000),
license_style VARCHAR(1000),
owner VARCHAR(1000),
SPDX_key VARCHAR(1000),
publication_year VARCHAR(4),
keywords VARCHAR(1000),
special_obligations VARCHAR(1000),
homepage_url VARCHAR(1000),
text_urls VARCHAR(1000),
osi_url VARCHAR(1000),
faq_url VARCHAR(1000),
guidance_url VARCHAR(1000),
other_urls VARCHAR(1000),
owner_name VARCHAR(1000),
owner_homepage_url VARCHAR(1000),
owner_type VARCHAR(1000),
contact_information VARCHAR(1000),
alias VARCHAR(1000),
owner_notes VARCHAR(1000)
);
CREATE TABLE nexb_license_requirements (
nexb_key VARCHAR(100),
requirements VARCHAR(200),
CONSTRAINT nexb_license_requirements_pk PRIMARY KEY (nexb_key,requirements)
);
CREATE TABLE nexb_requirements(
requirements VARCHAR(200) PRIMARY KEY,
description VARCHAR(1000) NOT NULL,
type VARCHAR(100) NOT NULL
 );
 ALTER TABLE nexb_license_requirements
 ADD CONSTRAINT nexb_license_requirements_fk FOREIGN KEY (requirements) REFERENCES nexb_requirements(requirements);
 
 CREATE TABLE olex_license(
 olex_key VARCHAR(100) PRIMARY KEY,
 name VARCHAR(200) NOT NULL,
 url VARCHAR(300) NOT NULL,
 model VARCHAR(100),
 class VARCHAR(100),
 license_base VARCHAR(200)
 );
 CREATE TABLE olex_obligation(
 olex_obligation_code VARCHAR(100) PRIMARY KEY,
 name VARCHAR(1000) NOT NULL,
 shortened VARCHAR(300));
 CREATE TABLE olex_license_obligation(
 olex_key VARCHAR(100) NOT NULL,
 olex_obligation_code VARCHAR(100) NOT NULL,
 CONSTRAINT olex_license_obligation_pk PRIMARY KEY (olex_key,olex_obligation_code),
 CONSTRAINT olex_obligation_fk FOREIGN KEY (olex_obligation_code) REFERENCES olex_obligation(olex_obligation_code),
 CONSTRAINT olex_key_fk FOREIGN KEY (olex_key) REFERENCES olex_license(olex_key)
 );
 CREATE TABLE match_license(
 olex_key VARCHAR(100)PRIMARY KEY,
 tldr_key VARCHAR(100),
 SPDX_key VARCHAR(100),
 nexb_key VARCHAR(100)
 );
